    # 继承的几种方式

    # 1.原型链继承

    原型链继承的主要思想是：重写子类的prototype属性，将其指向父类的实例

    function Animal() {
  this.kind = '我是动物';
}
Animal.prototype.eat = function () {
  console.log('我在吃饭');
};

function Cat(name, age) {
  this.name = name;
  this.age = age;
  this.say = function () {
    console.log(`我叫${this.name},我今年${this.age}岁了`);
  };
  this.run = function () {
    console.log('我在跑');
  };
}

Cat.prototype = new Animal();	// ★★★
Cat.prototype.constructor = Cat;	// ★★★

const mimi = new Cat('海参', 2);
    # 优点：
    1. 简单易于实现，只需要设置子类的prototype属性指向父类的实例即可
    2. 可通过子类直接访问父类的实例属性和原型链属性
    # 缺点：
    1. 子类的所有实例将共享父类的属性
    2. 创建子类实例时，不能向父类构造函数中传参数
    3. 无法实现多继承，子类的prototype只能设置一个值，设置多个值时，后面的值会覆盖前面的值

    # 2.构造函数继承

    构造函数继承的主要思想：在子类的构造函数中通过call()函数改变this的指向，调用父类的构造函数，从而将父类的实例的属性和函数绑定到子类的this上

    function Animal(kind) {
  this.kind = `我是${kind || '动物'}`;
}
Animal.prototype.eat = function () {
  console.log('我在吃饭');
};

function Cat(name, age, kind) {
  Animal.call(this, kind);	// ★★★
  this.name = name;
  this.age = age;
  this.say = function () {
    console.log(`我叫${this.name},我今年${this.age}岁了`);
  };
  this.run = function () {
    console.log('我在跑');
  };
}

const mimi = new Cat('海参', 2, '猫科动物');
    # 优点：
    1. 解决了子类实例共享父类属性的问题
    2. 创建子类的实例时，可以向父类传参
    3. 可以实现多继承
    # 缺点：
    1. 只能继承父类实例的属性和函数，并不能继承原型对象上的属性和函数
    2. 实例只是子类的实例，并不是父类的实例

    # 3.组合式继承

    组合继承的主要思想：结合构造继承和原型继承的两种方式，一方面在子类的构造函数中通过call()函数调用父类的构造函数，将父类的实例的属性和函数绑定到子类的this中；另一方面，通过改变子类的prototype属性，继承父类的原型对象上的属性和函数。

    function Animal(kind) {
  this.kind = `我是${kind || '动物'}`;
}
Animal.prototype.eat = function () {
  console.log('我在吃饭');
};

function Cat(name, age, kind) {
  Animal.call(this, kind);	// ★★★
  this.name = name;
  this.age = age;
  this.say = function () {
    console.log(`我叫${this.name},我今年${this.age}岁了`);
  };
  this.run = function () {
    console.log('我在跑');
  };
}

Cat.prototype = new Animal();	// ★★★
Cat.prototype.constructor = Cat;	// ★★★

const mimi = new Cat('海参', 2, '猫科动物');
    # 优点：
    1. 既能继承父类实例的属性和函数，又能继承原型对象上的属性和函数
    2. 不存在引用属性共享的问题
    3. 可以向父类的构造函数中传参
    # 缺点：
    1. 调用了两次父类构造函数，生成了两份实例（子类实例将子类原型上的那份屏蔽了）

    # 4.寄生组合继承

    组合继承方案已经足够好，但是针对其存在的缺点，我们仍然可以进行优化。

    在进行子类的prototype属性的设置时，可以去掉父类实例的属性的函数

    function Animal(kind) {
  this.kind = `我是${kind || '动物'}`;
}
Animal.prototype.eat = function () {
  console.log('我在吃饭');
};

function Cat(name, age, kind) {
  Animal.call(this, kind);	// ★★★
  this.name = name;
  this.age = age;
  this.say = function () {
    console.log(`我叫${this.name},我今年${this.age}岁了`);
  };
  this.run = function () {
    console.log('我在跑');
  };
}

Cat.prototype = Object.create(Animal.prototype);	// ★★★
Cat.prototype.constructor = Cat;	// ★★★

const mimi = new Cat('海参', 2, '猫科动物');
