# 元素水平垂直居中方法

# 一、利用margin: auto
# margin: auto表示自动填充剩余空间，且只会填充水平空间，不会填充垂直空间
# margin: auto生效的前提：当前必须是块级元素，并且设置宽高
# 设置position: absolute; top: 0; left: 0; right: 0; bottom: 0;，填满父元素宽高，用于垂直方向居中
 <style>
  .box {
    position: relative;
    width: 600px;
    height: 600px;
    background-color: pink;
  }

  .box1 {
    position: absolute;
    top: 0;
    left: 0;
    right: 0;
    bottom: 0;
    margin: auto;
    width: 100px;
    height: 100px;
    background-color: skyblue;
  }
</style>
    
<div class="box">
  <div class="box1"></div>
</div>
# 二、利用position（margin） 配合 transform
 <style>
  .box {
    position: relative;
    width: 600px;
    height: 600px;
    background-color: pink;
  }

  .box1 {
    position: absolute;
    top: 50%;  // margin-top: 50%
    left: 50%; // margin-left: 50%
    
    width: 100px;
    height: 100px;
    transform: translate(-50%, -50%);
    background-color: skyblue;
  }
</style>

 <div class="box">
  <div class="box1">box1</div>
</div>
# 三、利用flex布局
 <style>
  .box {
    display: flex;
    justify-content: center;
    align-items: center;
    width: 600px;
    height: 600px;
    background-color: pink;
  }

  .box1 {
    width: 100px;
    height: 100px;
    background-color: skyblue;
  }
</style>

<div class="box">
  <div class="box1">box1</div>
</div>
# 四、利用grid布局
 <style>
  .box {
    display: grid;
    width: 600px;
    height: 600px;
    background-color: pink;
  }

  .box1 {
    justify-self: center;
    align-self: center;
    width: 100px;
    height: 100px;
    background-color: skyblue;
  }
</style>

<div class="box">
  <div class="box1">box1</div>
</div>
